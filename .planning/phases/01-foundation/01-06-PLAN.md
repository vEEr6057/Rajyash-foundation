Pattern 5 — onboarding server action:
```typescript
// src/features/auth/actions/onboardingActions.ts
'use server';
import { auth, clerkClient } from '@clerk/nextjs/server';
import { onboardingSchema } from '../validations/onboarding.schema';

export async function completeOnboarding(input: unknown) {
  const { userId } = await auth();
  if (!userId) throw new Error('UNAUTHORIZED');

  const parsed = onboardingSchema.safeParse(input);
  if (!parsed.success) throw new Error('VALIDATION_ERROR');

  const client = await clerkClient();
  await client.users.updateUserMetadata(userId, {
    publicMetadata: {
      role: parsed.data.role,
      onboardingComplete: true,
    },
  });

  // Also upsert profile in DB (Phase 2+ joins against this)
  const { getDb } = await import('@/server/db');
  const db = getDb();
  const { profiles } = await import('@/server/db/schema');
  const { eq } = await import('drizzle-orm');
  await db.insert(profiles).values({
    id: userId,
    name: parsed.data.name,
    role: parsed.data.role,
    city: parsed.data.city ?? 'Ahmedabad',
    phone: parsed.data.phone,
    onboardingComplete: true,
  }).onConflictDoUpdate({
    target: profiles.id,
    set: {
      name: parsed.data.name,
      role: parsed.data.role,
      city: parsed.data.city ?? 'Ahmedabad',
      onboardingComplete: true,
      updatedAt: new Date(),
    },
  });
}
```

```typescript
import { z } from 'zod';

export const onboardingSchema = z.object({
  role: z.enum(['donor', 'volunteer'], {
    error: 'Please select whether you are a donor or volunteer',
  }),
  name: z.string().min(2, { message: 'Name must be at least 2 characters' }),
  city: z.string().default('Ahmedabad'),
  phone: z.string().optional(),
});

export type OnboardingFormData = z.infer<typeof onboardingSchema>;
```

```typescript
'use server';
import 'server-only';
import { auth, clerkClient } from '@clerk/nextjs/server';
import { eq } from 'drizzle-orm';
import { getDb } from '@/server/db';
import { profiles } from '@/server/db/schema';
import { onboardingSchema } from '../validations/onboarding.schema';

export async function completeOnboarding(input: unknown) {
  // AUTH-05: server action independently re-verifies session.
  // Never trust that middleware has already checked — CVE-2025-29927 showed middleware
  // can be bypassed. This is the real auth gate.
  const { userId } = await auth();
  if (!userId) throw new Error('UNAUTHORIZED');

  const parsed = onboardingSchema.safeParse(input);
  if (!parsed.success) throw new Error('VALIDATION_ERROR');

  // Update Clerk publicMetadata — this is the auth source of truth (D-02)
  const client = await clerkClient();
  await client.users.updateUserMetadata(userId, {
    publicMetadata: {
      role: parsed.data.role,
      onboardingComplete: true,
    },
  });

  // Upsert profile in DB — read-optimized copy for Phase 2+ joins (D-01 note in schema.ts)
  const db = getDb();
  await db
    .insert(profiles)
    .values({
      id: userId,
      name: parsed.data.name,
      role: parsed.data.role,
      city: parsed.data.city,
      phone: parsed.data.phone ?? null,
      onboardingComplete: true,
    })
    .onConflictDoUpdate({
      target: profiles.id,
      set: {
        name: parsed.data.name,
        role: parsed.data.role,
        city: parsed.data.city,
        onboardingComplete: true,
        updatedAt: new Date(),
      },
    });
}
```

```tsx
'use client';
import { type Control, type FieldValues, type Path, useController } from 'react-hook-form';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { cn } from '@/lib/cn';

interface FormFieldProps<T extends FieldValues> {
  control: Control<T>;
  name: Path<T>;
  label: string;
  placeholder?: string;
  type?: string;
  required?: boolean;
  disabled?: boolean;
}

export function FormField<T extends FieldValues>({
  control,
  name,
  label,
  placeholder,
  type = 'text',
  required,
  disabled,
}: FormFieldProps<T>) {
  const { field, fieldState } = useController({ name, control });
  const hasError = !!fieldState.error;

  return (
    <div className="flex flex-col gap-1.5">
      <Label htmlFor={String(name)} className={cn(required && 'after:content-["*"] after:ml-0.5 after:text-destructive')}>
        {label}
      </Label>
      <Input
        {...field}
        id={String(name)}
        type={type}
        placeholder={placeholder}
        disabled={disabled}
        aria-invalid={hasError}
        aria-describedby={hasError ? `${String(name)}-error` : undefined}
        className={cn(hasError && 'border-destructive focus-visible:ring-destructive')}
      />
      {hasError && (
        <p id={`${String(name)}-error`} className="text-destructive text-sm" role="alert">
          {fieldState.error?.message}
        </p>
      )}
    </div>
  );
}
```

```tsx
'use client';
import { type Control, type FieldValues, type Path, useController } from 'react-hook-form';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';
import { Label } from '@/components/ui/label';
import { cn } from '@/lib/cn';

interface SelectOption {
  value: string;
  label: string;
}

interface FormSelectProps<T extends FieldValues> {
  control: Control<T>;
  name: Path<T>;
  label: string;
  options: SelectOption[];
  placeholder?: string;
  required?: boolean;
  disabled?: boolean;
}

export function FormSelect<T extends FieldValues>({
  control,
  name,
  label,
  options,
  placeholder = 'Select an option',
  required,
  disabled,
}: FormSelectProps<T>) {
  const { field, fieldState } = useController({ name, control });
  const hasError = !!fieldState.error;

  return (
    <div className="flex flex-col gap-1.5">
      <Label htmlFor={String(name)} className={cn(required && 'after:content-["*"] after:ml-0.5 after:text-destructive')}>
        {label}
      </Label>
      <Select
        value={field.value}
        onValueChange={field.onChange}
        disabled={disabled}
      >
        <SelectTrigger
          id={String(name)}
          aria-invalid={hasError}
          className={cn(hasError && 'border-destructive')}

          {options.map((opt) => (
            <SelectItem key={opt.value} value={opt.value}>

      {hasError && (
```

```tsx
'use client';
// OnboardingForm — D-05: role-select (donor/volunteer) + name, city defaults to Ahmedabad.
// Admin is NOT offered (D-01). City is pre-filled; phone is optional.
import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { useRouter } from 'next/navigation';
import { useUser } from '@clerk/nextjs';
import { useState } from 'react';
import { FormField } from '@/components/forms/FormField';
import { FormSelect } from '@/components/forms/FormSelect';
import { Button } from '@/components/ui/button';
import { onboardingSchema, type OnboardingFormData } from '@/features/auth';
import { completeOnboarding } from '../actions/onboardingActions';
import { logger } from '@/lib/logger';

const ROLE_OPTIONS = [
  { value: 'donor', label: 'Food Donor — I donate surplus food' },
  { value: 'volunteer', label: 'Volunteer — I pick up and deliver food' },
];

export function OnboardingForm() {
  const { user } = useUser();
  const router = useRouter();
  const [error, setError] = useState<string | null>(null);

  const { control, handleSubmit, formState: { isSubmitting } } = useForm<OnboardingFormData>({
    resolver: zodResolver(onboardingSchema),
    defaultValues: {
      role: undefined,
      name: user?.fullName ?? '',
      city: 'Ahmedabad',
      phone: user?.primaryPhoneNumber?.phoneNumber ?? '',
    },
  });

  async function onSubmit(data: OnboardingFormData) {
    setError(null);
    try {
      await completeOnboarding(data);
      // CRITICAL: Must call user.reload() after metadata update.
      // Without this, the session JWT still carries the old (onboardingComplete: false) claims
      // and middleware redirects back to /onboarding (Pitfall 4 in RESEARCH.md).
      await user?.reload();
      router.push('/portal/dashboard');
    } catch (err) {
      logger.error('Onboarding failed:', err);
      setError('Something went wrong. Please try again.');
    }
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="flex flex-col gap-5">
      <FormSelect
        control={control}
        name="role"
        label="I want to"
        options={ROLE_OPTIONS}
        placeholder="Select your role"
        required
      />
      <FormField
        control={control}
        name="name"
        label="Full name"
        placeholder="Your name"
        required
      />
      <FormField
        control={control}
        name="city"
        label="City"
        placeholder="Ahmedabad"
        disabled  // Ahmedabad default, disabled for v1 (single-city per SUMMARY.md)
      />

      {error && (
        <p className="text-destructive text-sm" role="alert">{error}</p>
      )}

      <Button
        type="submit"
        disabled={isSubmitting}
        className="w-full bg-primary text-primary-foreground hover:bg-primary/90 rj-press"
      >
        {isSubmitting ? 'Setting up your account…' : 'Continue to dashboard'}
      </Button>
    </form>
  );
}
```
