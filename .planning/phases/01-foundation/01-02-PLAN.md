  variable: '--font-sans',
  weight: ['400', '500', '600', '700'],
  display: 'swap',

// Clerk appearance — themed to our saffron design tokens (per D-04)
const clerkAppearance = {
  variables: {
    colorPrimary: '#C04E12',        // --primary
    colorBackground: '#FFFFFF',     // --surface
    colorForeground: '#22271F',     // --foreground
    colorNeutral: '#6E665A',        // --muted-foreground
    colorDanger: '#BE3B2E',         // --destructive
    colorBorder: '#E8E0CF',         // --border
    colorInput: '#FFFFFF',
    borderRadius: '0.75rem',        // --radius (12px)
    fontFamily: '"Mukta", system-ui, sans-serif',
  },
  elements: {
    formButtonPrimary: 'bg-primary text-primary-foreground hover:bg-primary/90 text-sm font-medium',
    card: 'shadow-md',
  },

export const metadata: Metadata = {
  title: 'Rajyash Food Rescue',
  description: 'Rescuing surplus food, with warmth.',
};

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;

  return (

      <head>
        {/* Noto Sans Gujarati — not available in next/font/google; loaded via <link> */}
        {/* Noto Sans Devanagari covered by Mukta's devanagari subset */}

NOTE: QueryClientProvider is wired here in Phase 1 as a shell (required for Phase 2+ hooks).
Add it as a client component wrapper after ClerkProvider:

Create `src/components/providers/QueryProvider.tsx`:
```tsx
'use client';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { useState } from 'react';

export function QueryProvider({ children }: { children: React.ReactNode }) {
  const [queryClient] = useState(() => new QueryClient({
    defaultOptions: { queries: { retry: false, staleTime: 30_000 } },
  }));
  return <QueryClientProvider client={queryClient}>{children}</QueryClientProvider>;
}
```
