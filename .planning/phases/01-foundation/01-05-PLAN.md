```typescript
// src/middleware.ts
import { clerkMiddleware, createRouteMatcher } from '@clerk/nextjs/server';
import { NextResponse } from 'next/server';

const isPublicRoute = createRouteMatcher([
  '/',
  '/sign-in(.*)',
  '/sign-up(.*)',
  '/api/webhook(.*)',
]);
const isOnboardingRoute = createRouteMatcher(['/onboarding(.*)']);
const isAdminRoute = createRouteMatcher(['/admin(.*)', '/(admin)(.*)']);
const isPortalRoute = createRouteMatcher(['/(portal)(.*)']);

export default clerkMiddleware(async (auth, req) => {
  const { isAuthenticated, sessionClaims, redirectToSignIn } = await auth();

  if (isPublicRoute(req)) return NextResponse.next();
  if (!isAuthenticated) return redirectToSignIn({ returnBackUrl: req.url });

  if (!sessionClaims?.metadata?.onboardingComplete && !isOnboardingRoute(req)) {
    return NextResponse.redirect(new URL('/onboarding', req.url));
  }

  if (isAdminRoute(req) && sessionClaims?.metadata?.role !== 'admin') {
    return new NextResponse(null, { status: 403 });
  }

  return NextResponse.next();
});

export const config = {
  matcher: [
    '/((?!_next|[^?]*\\.(?:html?|css|js(?!on)|jpe?g|webp|png|gif|svg|ttf|woff2?|ico|csv|docx?|xlsx?|zip|webmanifest)).*)',
    '/(api|trpc)(.*)',
    '/__clerk/(.*)',
  ],
};
```

Pattern 2 — requireRole() helper:
```typescript
// src/server/auth/session.ts
import 'server-only';
import { auth } from '@clerk/nextjs/server';
import type { Role } from '@/features/auth';

export async function requireRole(allowedRoles: Role[]) {
  const { userId, sessionClaims } = await auth();
  if (!userId) throw new Error('UNAUTHORIZED');
  const role = sessionClaims?.metadata?.role as Role | undefined;
  if (!role || !allowedRoles.includes(role)) throw new Error('FORBIDDEN');
  return { userId, role };
}
```

- requireRole() in every server action is the defense-in-depth layer
- Never trust middleware headers from external sources

import { requireRole } from './session';

const mockAuth = auth as ReturnType<typeof vi.fn>;

describe('requireRole', () => {
  beforeEach(() => {
    vi.resetAllMocks();
  });

  it('returns { userId, role } for an authenticated user with an allowed role', async () => {
    mockAuth.mockResolvedValue({
      userId: 'user_test123',
      sessionClaims: { metadata: { role: 'donor', onboardingComplete: true } },

    const result = await requireRole(['donor']);
    expect(result).toEqual({ userId: 'user_test123', role: 'donor' });

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Internet → Next.js middleware | Every non-static request passes through clerkMiddleware |
| Middleware → server actions | Middleware checks are a first gate; server actions re-verify independently (AUTH-05) |
| Client → server actions | Server actions are public POST endpoints; requireRole() is the auth boundary |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Spoofing | Middleware bypass (CVE-2025-29927) | mitigate | Using Next.js ≥15.2.3 (15.5.x scaffolded in Plan 01-01). requireRole() in all server actions provides defense-in-depth — even if middleware is bypassed, actions still re-verify |
| T-05-02 | Elevation of Privilege | Role read from request body | mitigate | requireRole() reads role ONLY from `sessionClaims?.metadata?.role` via Clerk's auth(). Never from req.body, headers, or query params |
| T-05-03 | Elevation of Privilege | Admin role self-assignment | mitigate | D-01: admin role is never self-selectable. The onboarding form (Plan 01-06) only offers donor/volunteer. Admin assigned manually by an existing admin via Clerk Dashboard |
| T-05-04 | Spoofing | OTP flooding / SMS pumping | mitigate | AUTH-03: Clerk enforces IP-level rate limits (5 req/10s create, 3 req/10s attempt, 100-attempt lockout). Per-phone limiting is Phase 1 known gap — acceptable at NGO scale |
| T-05-05 | Information Disclosure | 403 vs redirect on wrong role | mitigate | Admin routes return explicit 403 response (not redirect). This prevents information leakage about portal route existence and avoids infinite redirect loops |
| T-05-06 | Tampering | IDOR in portal dashboard | mitigate | DashboardPage reads role from sessionClaims (server-side), not from URL params. No user-controlled input determines what data is shown |
</threat_model>
