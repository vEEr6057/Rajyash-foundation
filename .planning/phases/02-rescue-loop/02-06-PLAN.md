```typescript
'use client'

import { useTransition, useState } from 'react'
import { claimPickupAction } from '@/features/pickups'
import { useRouter } from 'next/navigation'

interface Props {
  pickupId:      string
  /** Injected for testing — allows tests to pre-set state without triggering real action */
  isPending?:    boolean
  error?:        string
  alreadyClaimed?: boolean
}

/**
 * Claim button with loading/retry/alreadyClaimed UX (D-10, VOL-03, VOL-06).
 *
 * State machine:
 *   idle → pending (claim in-flight) → success (redirects to board)
 *                                    → alreadyClaimed ("Just taken")
 *                                    → error ("Try again")
 */
export function ClaimButton({ pickupId, isPending: externalPending, error: externalError, alreadyClaimed: externalClaimed }: Props) {
  const router                      = useRouter()
  const [isPending, startTransition] = useTransition()
  const [state, setState]           = useState<
    | { type: 'idle' }
    | { type: 'alreadyClaimed'; message: string }
    | { type: 'error'; message: string }
  >({ type: 'idle' })

  // Allow tests to inject state via props
  const isLoading    = externalPending ?? isPending
  const errorMessage = externalError   ?? (state.type === 'error' ? state.message : null)
  const isAlreadyClaimed = externalClaimed ?? state.type === 'alreadyClaimed'

  async function handleClaim() {
    setState({ type: 'idle' })
    startTransition(async () => {
      const result = await claimPickupAction(pickupId)
      if (result.success) {
        router.refresh()
      } else if (result.alreadyClaimed) {
        setState({ type: 'alreadyClaimed', message: result.error })
      } else {
        setState({ type: 'error', message: result.error })
      }
    })
  }

  if (isAlreadyClaimed) {
    return (
      <div className="space-y-1">
        <p className="text-sm text-warning font-medium">
          {state.type === 'alreadyClaimed' ? state.message : 'Just taken by another volunteer.'}
        </p>
        <button
          type="button"
          onClick={() => { setState({ type: 'idle' }); router.refresh() }}
          className="rj-press text-sm text-primary underline"
        >
          Refresh list
        </button>
      </div>
    )
  }

  if (errorMessage) {
    return (
      <div className="space-y-1">
        <p className="text-xs text-danger">{errorMessage}</p>
        <button
          type="button"
          onClick={() => { setState({ type: 'idle' }); handleClaim() }}
          className="rj-press inline-flex h-[38px] items-center rounded-[10px]
            bg-primary px-4 text-sm font-semibold text-primary-foreground"
          aria-disabled={isLoading}
          disabled={isLoading}
        >
          Try again
        </button>
      </div>
    )
  }

  return (
    <button
      type="button"
      onClick={handleClaim}
      disabled={isLoading}
      aria-disabled={isLoading}
      className="rj-press inline-flex h-[38px] items-center gap-2 rounded-[10px]
        bg-primary px-4 text-sm font-semibold text-primary-foreground
        disabled:opacity-60 transition-opacity"
    >
      {isLoading && (
        <span
          className="h-3.5 w-3.5 rounded-full border-2 border-white/30 border-t-white animate-spin"
          aria-hidden="true"
        />
      )}
      {isLoading ? 'Claiming…' : 'Claim pickup'}
    </button>
  )
}
```

```typescript
'use client'

import { useTransition, useState } from 'react'
import { advanceStatusAction } from '@/features/pickups'
import { useRouter } from 'next/navigation'
import { PhotoUploader } from './PhotoUploader'
import type { PickupStatus } from '@/config/constants'

interface Props {
  pickupId: string
  toStatus: PickupStatus
  label:    string          // e.g. "Mark En Route", "Mark Picked Up", "Mark Delivered"
}

const NEEDS_PHOTO: PickupStatus[] = ['delivered']

export function StatusAdvanceButton({ pickupId, toStatus, label }: Props) {
  const router                       = useRouter()
  const [isPending, startTransition] = useTransition()
  const [error, setError]            = useState<string | null>(null)
  const [photoReady, setPhotoReady]  = useState(!NEEDS_PHOTO.includes(toStatus))

  async function handleAdvance() {
    if (!photoReady) return   // guard: require photo before advance to delivered
    setError(null)
    startTransition(async () => {
      const result = await advanceStatusAction(pickupId, toStatus)
      if (result.success) {
        router.refresh()
      } else {
        setError(result.error)
      }
    })
  }

  return (
    <div className="space-y-3">
      {/* Proof photo uploader — only shown before 'delivered' advance */}
      {NEEDS_PHOTO.includes(toStatus) && !photoReady && (
        <div className="space-y-2">
          <p className="text-sm font-medium text-foreground">
            Upload a proof-of-delivery photo first (VOL-05)
          </p>
          <PhotoUploader
            pickupId={pickupId}
            photoType="proof"
            onSuccess={() => setPhotoReady(true)}
          />
        </div>
      )}

      {error && (
        <p className="text-xs text-danger">{error}</p>
      )}

      <button
        type="button"
        onClick={handleAdvance}
        disabled={isPending || !photoReady}
        aria-disabled={isPending || !photoReady}
        className="rj-press inline-flex h-[46px] w-full items-center justify-center gap-2
          rounded-[10px] bg-leaf px-4 text-sm font-semibold text-white
          disabled:opacity-60 transition-opacity"
      >
        {isPending && (
          <span
            className="h-3.5 w-3.5 rounded-full border-2 border-white/30 border-t-white animate-spin"
            aria-hidden="true"
          />
        )}
        {isPending ? 'Saving…' : label}
      </button>

      {error && (
        <button
          type="button"
          onClick={() => { setError(null); handleAdvance() }}
          className="rj-press text-sm text-primary underline w-full text-center"
        >
          Retry
        </button>
      )}
    </div>
  )
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Volunteer browser → claimPickupAction | pickupId from URL/prop; actorId from Clerk session; race is handled server-side |
| Volunteer browser → advanceStatusAction | toStatus from button label; ownership + transition validated server-side |
| Volunteer browser → PhotoUploader → Storage | PUT to Supabase signed URL directly; path server-constructed in getSignedUploadUrlAction |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-22 | Elevation of Privilege | Double-claim race — two volunteers tap Claim simultaneously | mitigate | Atomic conditional UPDATE (D-07) — exactly one volunteer wins; loser gets `alreadyClaimed: true` response; ClaimButton UI shows "Just taken" |
| T-02-23 | Tampering | Volunteer advances another volunteer's pickup via crafted advanceStatusAction call | mitigate | `advanceStatusAction` → `pickupService.advanceStatus` checks `pickup.volunteerId === actorId`; throws ForbiddenError if mismatch |
| T-02-24 | Tampering | Volunteer sends `toStatus='delivered'` skipping intermediate states | mitigate | `VALID_TRANSITIONS` map + `validateTransition` in service layer; invalid transitions are rejected before DB write |
| T-02-25 | Denial of Service | Board page caches stale state after a claim | mitigate | `export const dynamic = 'force-dynamic'` on board page + `revalidatePath('/portal/board')` in `claimPickupAction` |
| T-02-26 | Spoofing | Fake proof-of-delivery photo upload to wrong bucket path | mitigate | `getSignedUploadUrlAction` constructs path server-side; `saveProofPhotoPath` checks `volunteerId === actorId` before DB update |
</threat_model>
