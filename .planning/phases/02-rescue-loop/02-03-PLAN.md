```typescript
/**
 * Supabase Storage service-role client — SERVER ONLY.
 *
 * The service-role key bypasses Supabase RLS and must never reach the browser.
 * `import 'server-only'` enforces this at build time (throws if imported client-side).
 *
 * Usage in server actions / RSC:
 *   import { supabaseStorage } from '@/lib/storage'
 *   const { data, error } = await supabaseStorage
 *     .from('pickups')
 *     .createSignedUploadUrl(objectPath)
 */
import 'server-only'
import { createClient } from '@supabase/supabase-js'
import { env } from '@/config/env'

// Create a storage-only client with the service-role key.
// auth.autoRefreshToken + persistSession disabled — this is a server-side client,
// not a user session. The service-role key never expires.
export const supabaseStorage = createClient(
  env.SUPABASE_URL,
  env.SUPABASE_SERVICE_ROLE_KEY,
  {
    auth: {
      autoRefreshToken: false,
      persistSession:   false,
    },
  },
).storage

/**
 * Generate a signed upload URL for a pickup photo (DON-02, D-04).
 * Call from a server action — never expose this function to the client.
 *
 * The browser uploads directly to Supabase Storage via a PUT to the returned URL
 * (no Next.js worker involved — reduces latency and Cloudflare egress).
 *
 * @param objectPath  e.g. 'pickups/abc-123/user-id-1719398400000.jpg'
 * @returns signedUrl (PUT target), token (for supabase-js uploadToSignedUrl), objectPath
 */
export async function createSignedUploadUrl(objectPath: string): Promise<{
  signedUrl: string
  token: string
  objectPath: string
}> {
  const { data, error } = await supabaseStorage
    .from('pickups')
    .createSignedUploadUrl(objectPath)

  if (error || !data) {
    throw new Error(`Storage: could not create signed upload URL — ${error?.message ?? 'unknown'}`)
  }

  return {
    signedUrl: data.signedUrl,
    token:     data.token,
    objectPath,
  }
}

/**
 * Generate a short-lived signed download URL for displaying a pickup photo (D-03, D-04).
 * Call from a server action or RSC where the image URL is needed.
 * Default expiry: 1 hour. For proof-of-delivery photos, 24 hours may be preferred.
 *
 * @param objectPath  the path stored in the DB (e.g. 'pickups/abc-123/...')
 * @param expiresInSeconds  default 3600 (1 hour)
 * @returns signed download URL
 */
export async function createSignedDownloadUrl(
  objectPath: string,
  expiresInSeconds = 3600,
): Promise<string> {
  const { data, error } = await supabaseStorage
    .from('pickups')
    .createSignedUrl(objectPath, expiresInSeconds)

  if (error || !data) {
    throw new Error(`Storage: could not create signed download URL — ${error?.message ?? 'unknown'}`)
  }

  return data.signedUrl
}
```

```typescript
/**
 * Nominatim/OpenStreetMap geocoder — SERVER ONLY (D-05).
 *
 * Called ONCE per pickup creation. Result (lat, lng) stored in the DB.
 * Never call this at render time — Nominatim rate limit is 1 req/s.
 *
 * Usage policy requirements (nominatim.org/policies/):
 *  1. User-Agent must identify the application + contact email.
 *  2. Max 1 request/second (irrelevant at NGO scale — creation-time only).
 *  3. No bulk geocoding / crawling.
 */
import 'server-only'

export interface GeoResult {
  lat:         number   // WGS84 latitude
  lng:         number   // WGS84 longitude (Nominatim response field is 'lon', not 'lng')
  displayName: string   // Human-readable address returned by Nominatim
}

// Nominatim response shape (fields we care about — response has more)
interface NominatimResult {
  lat:          string   // Note: strings, not numbers — parse with parseFloat()
  lon:          string   // Note: 'lon' not 'lng'
  display_name: string
}

/**
 * Geocode an address string to WGS84 lat/lng using Nominatim/OSM.
 *
 * Appends "Ahmedabad, India" to the query to bias results to our city.
 * Returns null if:
 *  - No results found (address not recognised)
 *  - Network error or non-200 response
 *
 * When null is returned, the form should surface:
 * "Address not found — please drag the pin to set your location."
 * The Leaflet draggable pin is the user's escape hatch (D-05).
 */
export async function geocodeAddress(address: string): Promise<GeoResult | null> {
  // Bias to Ahmedabad, India to improve accuracy for our single-city scope.
  const query = encodeURIComponent(`${address}, Ahmedabad, India`)
  const url   = `https://nominatim.openstreetmap.org/search?q=${query}&format=json&limit=1&addressdetails=0`

  let res: Response
  try {
    res = await fetch(url, {
      headers: {
        // Required by Nominatim usage policy — identify app + contact.
        'User-Agent':      'RajyashFoodRescue/1.0 (rajyashfoundation@rajyashgroup.com)',
        'Accept-Language': 'en',
        'Accept':          'application/json',
      },
      // Workers: disable Next.js fetch cache for this call (creation-time, no caching wanted)
      cache: 'no-store',
    })
  } catch {
    // Network failure (Cloudflare edge can't reach Nominatim) — return null gracefully
    return null
  }

  if (!res.ok) return null

  let results: NominatimResult[]
  try {
    results = (await res.json()) as NominatimResult[]
  } catch {
    return null
  }

  if (!results.length) return null

  const first = results[0]
  return {
    lat:         parseFloat(first.lat),  // Nominatim returns strings — must parseFloat
    lng:         parseFloat(first.lon),  // Note: 'lon' in Nominatim, 'lng' in our schema
    displayName: first.display_name,
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| server action → storage.ts | Service-role key stays in server action context; never serialised to response |
| storage.ts → Supabase Storage CDN | HTTPS only; signed URLs expire (1h download, configurable upload) |
| server action → geocoding.ts → Nominatim | Geocoding is auth-gated (requireRole(['donor']) in server action); no unauthenticated access |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-09 | Information Disclosure | `SUPABASE_SERVICE_ROLE_KEY` leak via client bundle | mitigate | `import 'server-only'` at top of storage.ts — build fails if imported from a `"use client"` component. Never export `supabaseStorage` from a barrel that a client component imports. |
| T-02-10 | Information Disclosure | Signed download URL scope creep — URL shared beyond intended recipient | accept | URLs expire in 1h (short-lived); bucket is private; no sensitive PII in images at NGO scale. Upgrade path: per-request signed URLs (already the pattern). |
| T-02-11 | Denial of Service | Nominatim abuse — geocoding spammed via pickup creation | mitigate | `geocodeAddress` called only inside `createPickupAction` which is behind `requireRole(['donor'])`; auth gate throttles to one authenticated donor at a time at NGO scale. |
