// Zod v4 syntax: { error: '...' } not { required_error: '...' }
// min/max errors use { message: '...' }

export const createPickupSchema = z.object({
  foodCategory: z.enum(
    ['cooked_meal', 'raw_produce', 'packaged', 'bakery', 'other'],
    { error: 'Please select a food category' },
  ),
  description: z
    .string()
    .min(10, { message: 'Please describe the food (min 10 characters)' }),
  quantity: z.number().int().min(1, { message: 'Quantity must be at least 1' }),
  quantityUnit: z.enum(['servings', 'kg'], { error: 'Unit must be servings or kg' }),
  address: z.string().min(5, { message: 'Please enter a full address' }),
  windowStart: z.string().datetime({ message: 'Invalid pickup window start time' }),
  windowEnd: z.string().datetime({ message: 'Invalid pickup window end time' }),
  safetyAttested: z.literal(true, {
    error: 'You must confirm the food safety attestation',
  }),
  // lat/lng are filled server-side by geocoding; not in the form schema
  lat: z.number().optional(),
  lng: z.number().optional(),
})

// Partial schema for donor edit (DON-05): only changed fields required.
// safetyAttested and foodCategory can be re-asserted if donor edits them.
export const updatePickupSchema = createPickupSchema
  .omit({ safetyAttested: true })
  .partial()
  .extend({
    safetyAttested: z
      .literal(true, { error: 'You must confirm the food safety attestation' })
      .optional(),
  })

export type CreatePickupInput = z.infer<typeof createPickupSchema>
export type UpdatePickupInput = z.infer<typeof updatePickupSchema>

import 'server-only'
import { and, desc, eq, sql } from 'drizzle-orm'
import { getDb } from '@/server/db/client'
import { pickups, type NewPickup, type Pickup } from '@/server/db/schema'
import type { PickupStatus } from '@/config/constants'

/** Thin Drizzle data-access layer for pickups. No business logic — that lives in pickupService. */
export const pickupsRepo = {
  async getById(id: string): Promise<Pickup | null> {
    const db = getDb()
    const rows = await db.select().from(pickups).where(eq(pickups.id, id)).limit(1)
    return rows[0] ?? null
  },

  /** All open (status=requested) pickups for the volunteer board (VOL-01). Newest first. */
  async getOpen(): Promise<Pickup[]> {
    const db = getDb()
    return db
      .select()
      .from(pickups)
      .where(eq(pickups.status, 'requested'))
      .orderBy(desc(pickups.createdAt))
  },

  /** Donor's own pickups (DON-04). Newest first. */
  async getByDonorId(donorId: string): Promise<Pickup[]> {
    const db = getDb()
    return db
      .select()
      .from(pickups)
      .where(eq(pickups.donorId, donorId))
      .orderBy(desc(pickups.createdAt))
  },

  async insert(input: NewPickup): Promise<Pickup> {
    const db = getDb()
    const rows = await db.insert(pickups).values(input).returning()
    return rows[0]
  },

  /** General-purpose update (edit form — restricted to status=requested in service). */
  async update(id: string, data: Partial<NewPickup>): Promise<Pickup | null> {
    const db = getDb()
    const rows = await db
      .update(pickups)
      .set({ ...data, updatedAt: sql`now()` })
      .where(eq(pickups.id, id))
      .returning()
    return rows[0] ?? null
  },

  /** Status-only update — used by advanceStatus and cancelPickup after validation. */
  async updateStatus(id: string, status: PickupStatus): Promise<Pickup | null> {
    const db = getDb()
    const rows = await db
      .update(pickups)
      .set({ status, updatedAt: sql`now()` })
      .where(eq(pickups.id, id))
      .returning()
    return rows[0] ?? null
  },

  /**
   * Atomic conditional claim (D-07, VOL-03).
   * UPDATE WHERE status='requested' RETURNING — exactly one winner in a race.
   * Returns the claimed row on success, null if 0 rows updated (already claimed).
   */
  async claimIfAvailable(pickupId: string, volunteerId: string): Promise<Pickup | null> {
    const db = getDb()
    const rows = await db
      .update(pickups)
      .set({
        status:      'accepted',
        volunteerId,
        claimedAt:   sql`now()`,
        updatedAt:   sql`now()`,
      })
      .where(
        and(
          eq(pickups.id, pickupId),
          eq(pickups.status, 'requested'),
        ),
      )
      .returning()
    return rows[0] ?? null
  },
}

**File 1: `src/features/pickups/services/pickupService.ts`**

```typescript
import 'server-only'
import { pickupsRepo } from '@/server/db/repositories/pickups'
import { statusEventsRepo } from '@/server/db/repositories/statusEvents'
import { VALID_TRANSITIONS, type PickupStatus } from '@/config/constants'
import type { Pickup } from '@/server/db/schema'
import type { UpdatePickupInput } from '../validations/pickup.schema'

// ─── Domain errors ────────────────────────────────────────────────────────────

export class PickupAlreadyClaimedError extends Error {
  constructor(msg = 'This pickup was just taken by another volunteer.') {
    super(msg); this.name = 'PickupAlreadyClaimedError'
  }
}

export class InvalidTransitionError extends Error {
  constructor(msg: string) {
    super(msg); this.name = 'InvalidTransitionError'
  }
}

export class ForbiddenError extends Error {
  constructor(msg: string) {
    super(msg); this.name = 'ForbiddenError'
  }
}

export class NotFoundError extends Error {
  constructor(msg = 'Pickup not found.') {
    super(msg); this.name = 'NotFoundError'
  }
}

// ─── Pure helper (also exported for tests) ───────────────────────────────────

/**
 * Validates a status transition against VALID_TRANSITIONS.
 * Throws InvalidTransitionError if the transition is not allowed.
 * This is a pure function — no DB access.
 */
export function validateTransition(from: PickupStatus, to: PickupStatus): void {
  if (!VALID_TRANSITIONS[from].includes(to)) {
    throw new InvalidTransitionError(
      `Cannot transition from '${from}' to '${to}'.`,
    )
  }
}

// ─── Service ─────────────────────────────────────────────────────────────────

export const pickupService = {
  /**
   * Atomic race-safe claim (D-07, VOL-03).
   * Returns the claimed pickup row on success.
   * Throws PickupAlreadyClaimedError if another volunteer claimed first.
   */
  async claimPickup(pickupId: string, volunteerId: string): Promise<Pickup> {
    const claimed = await pickupsRepo.claimIfAvailable(pickupId, volunteerId)
    if (!claimed) throw new PickupAlreadyClaimedError()

    await statusEventsRepo.insert({
      pickupId,
      fromStatus: 'requested',
      toStatus:   'accepted',
      actorId:    volunteerId,
    })

    return claimed
  },

  /**
   * Advance a claimed pickup through the status machine (D-08, VOL-04).
   * Only the assigned volunteer may advance past 'accepted'.
   * Actor check + transition validation both happen before any DB write.
   */
  async advanceStatus(
    pickupId: string,
    toStatus:  PickupStatus,
    actorId:   string,
  ): Promise<Pickup> {
    const pickup = await pickupsRepo.getById(pickupId)
    if (!pickup) throw new NotFoundError()

    // Actor ownership (AUTH-05 carried forward)
    if (pickup.volunteerId !== actorId) {
      throw new ForbiddenError('Only the assigned volunteer may advance this pickup.')
    }

    // Transition validity (D-08)
    validateTransition(pickup.status, toStatus)

    const updated = await pickupsRepo.updateStatus(pickupId, toStatus)
    if (!updated) throw new NotFoundError()

    await statusEventsRepo.insert({
      pickupId,
      fromStatus: pickup.status,
      toStatus,
      actorId,
    })

    return updated
  },

  /**
   * Donor cancels a pickup (D-08, D-09, DON-05).
   * Only the donor may cancel, and only while status === 'requested'.
   */
  async cancelPickup(pickupId: string, actorId: string): Promise<Pickup> {
    const pickup = await pickupsRepo.getById(pickupId)
    if (!pickup) throw new NotFoundError()

    if (pickup.donorId !== actorId) {
      throw new ForbiddenError('Only the donor may cancel this pickup.')
    }
    if (pickup.status !== 'requested') {
      throw new ForbiddenError(
        'A pickup can only be cancelled while it is in the requested state.',
      )
    }

    const updated = await pickupsRepo.updateStatus(pickupId, 'cancelled')
    if (!updated) throw new NotFoundError()

    await statusEventsRepo.insert({
      pickupId,
      fromStatus: 'requested',
      toStatus:   'cancelled',
      actorId,
    })

    return updated
  },

  /**
   * Donor edits a pickup that has not yet been claimed (D-09, DON-05).
   * Only the donor may edit, and only while status === 'requested'.
   */
  async editPickup(
    pickupId: string,
    actorId:  string,
    updates:  UpdatePickupInput,
  ): Promise<Pickup> {
    const pickup = await pickupsRepo.getById(pickupId)
    if (!pickup) throw new NotFoundError()

    if (pickup.donorId !== actorId) {
      throw new ForbiddenError('Only the donor may edit this pickup.')
    }
    if (pickup.status !== 'requested') {
      throw new ForbiddenError('A pickup can only be edited while it is in the requested state.')
    }

    const updated = await pickupsRepo.update(pickupId, updates as Record<string, unknown>)
    if (!updated) throw new NotFoundError()
    return updated
  },

  /**
   * Quick-repost: clone a prior pickup into a new 'requested' row (D-09, DON-06).
   * Returns the new pickup's id.
   */
  async repostPickup(originalPickupId: string, actorId: string): Promise<string> {
    const original = await pickupsRepo.getById(originalPickupId)
    if (!original) throw new NotFoundError()

    if (original.donorId !== actorId) {
      throw new ForbiddenError('You can only repost your own pickups.')
    }

    const newPickup = await pickupsRepo.insert({
      donorId:        original.donorId,
      volunteerId:    null,
      status:         'requested',
      foodCategory:   original.foodCategory,
      description:    original.description,
      quantity:       original.quantity,
      quantityUnit:   original.quantityUnit,
      safetyAttested: original.safetyAttested,
      address:        original.address,
      lat:            original.lat,
      lng:            original.lng,
      windowStart:    original.windowStart,
      windowEnd:      original.windowEnd,
      foodPhotoPath:  null,    // fresh request — no photo carried over
      proofPhotoPath: null,
      claimedAt:      null,
    })

    return newPickup.id
  },

  /** Update the food photo path after upload (DON-02). */
  async saveFoodPhotoPath(pickupId: string, actorId: string, path: string): Promise<void> {
    const pickup = await pickupsRepo.getById(pickupId)
    if (!pickup) throw new NotFoundError()
    if (pickup.donorId !== actorId) throw new ForbiddenError('Not your pickup.')
    await pickupsRepo.update(pickupId, { foodPhotoPath: path })
  },

  /** Update the proof photo path after volunteer upload (VOL-05). */
  async saveProofPhotoPath(pickupId: string, actorId: string, path: string): Promise<void> {
    const pickup = await pickupsRepo.getById(pickupId)
    if (!pickup) throw new NotFoundError()
    if (pickup.volunteerId !== actorId) throw new ForbiddenError('Not your assigned pickup.')
    await pickupsRepo.update(pickupId, { proofPhotoPath: path })
  },
}
```
