```typescript
// ─── Phase 2 enums ───────────────────────────────────────────────────────────

export const pickupStatusEnum = pgEnum('pickup_status', [
  'requested', 'accepted', 'en_route', 'picked_up', 'delivered', 'cancelled',
])

export const foodCategoryEnum = pgEnum('food_category', [
  'cooked_meal', 'raw_produce', 'packaged', 'bakery', 'other',
])

export const quantityUnitEnum = pgEnum('quantity_unit', ['servings', 'kg'])

// ─── pickups ─────────────────────────────────────────────────────────────────

/**
 * Core entity for the rescue loop. One row per food-rescue request.
 * PK is a UUID so IDs are unguessable (IDOR protection).
 */
export const pickups = pgTable('pickups', {
  id:             uuid('id').primaryKey().defaultRandom(),

  // Actors
  donorId:        text('donor_id').notNull().references(() => profiles.id),
  volunteerId:    text('volunteer_id').references(() => profiles.id),   // null until claimed

  // Status machine (D-08)
  status:         pickupStatusEnum('status').notNull().default('requested'),

  // Food details (D-01, D-02)
  foodCategory:   foodCategoryEnum('food_category').notNull(),
  description:    text('description').notNull(),
  quantity:       integer('quantity').notNull(),
  quantityUnit:   quantityUnitEnum('quantity_unit').notNull(),

  // Safety attestation (DON-03) — stored server-side; never trust client value alone
  safetyAttested: boolean('safety_attested').notNull().default(false),

  // Location (D-05) — geocoded once at creation; never recalculated at render
  address:        text('address').notNull(),
  lat:            doublePrecision('lat').notNull(),
  lng:            doublePrecision('lng').notNull(),

  // Time window (DON-01)
  windowStart:    timestamp('window_start', { withTimezone: true }).notNull(),
  windowEnd:      timestamp('window_end',   { withTimezone: true }).notNull(),

  // Photos — object path in Supabase 'pickups' bucket (D-03). NOT a public URL.
  foodPhotoPath:  text('food_photo_path'),
  proofPhotoPath: text('proof_photo_path'),   // set by volunteer at delivery (VOL-05)

  // Claim timestamp (populated by claimIfAvailable, D-07)
  claimedAt:      timestamp('claimed_at', { withTimezone: true }),

  createdAt:      timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
  updatedAt:      timestamp('updated_at', { withTimezone: true }).notNull().defaultNow(),
})

export type Pickup    = typeof pickups.$inferSelect
export type NewPickup = typeof pickups.$inferInsert

// ─── status_events (append-only audit log, D-08) ─────────────────────────────

/**
 * Immutable audit trail of every status transition.
 * Insert-only — never update or delete rows in this table.
 */
export const statusEvents = pgTable('status_events', {
  id:         uuid('id').primaryKey().defaultRandom(),
  pickupId:   uuid('pickup_id').notNull().references(() => pickups.id),
  fromStatus: pickupStatusEnum('from_status').notNull(),
  toStatus:   pickupStatusEnum('to_status').notNull(),
  actorId:    text('actor_id').notNull().references(() => profiles.id),
  note:       text('note'),
  occurredAt: timestamp('occurred_at', { withTimezone: true }).notNull().defaultNow(),
})

export type StatusEvent    = typeof statusEvents.$inferSelect
export type NewStatusEvent = typeof statusEvents.$inferInsert
```

    expect(VALID_TRANSITIONS['requested']).toContain('accepted')
  })

  it('allows accepted → en_route', () => {
    expect(VALID_TRANSITIONS['accepted']).toContain('en_route')
  })

  it('does not allow requested → delivered (skipping states)', () => {
    expect(VALID_TRANSITIONS['requested']).not.toContain('delivered')
  })

  it('does not allow delivered → any status (terminal state)', () => {
    expect(VALID_TRANSITIONS['delivered']).toHaveLength(0)
  })
})

describe('validateTransition', () => {
  it('throws InvalidTransitionError for an invalid transition', () => {
    expect(() => validateTransition('requested', 'delivered')).toThrow(InvalidTransitionError)

  it('does not throw for a valid transition', () => {
    expect(() => validateTransition('accepted', 'en_route')).not.toThrow()

})
