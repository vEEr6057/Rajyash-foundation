'use server'

/**
 * Pickup server actions — all mutations for the rescue loop.
 * Pattern (AUTH-05):
 *   1. requireRole([...]) — first line, every action. Throws on unauth/forbidden.
 *   2. Parse/validate input with Zod safeParse.
 *   3. Delegate to PickupService (business logic + ownership checks).
 *   4. revalidatePath on affected routes.
 *   5. Return serialisable result (no Prisma/Drizzle objects — use plain data).
 *
 * claimPickupAction is the exception: it returns a typed result object instead of
 * throwing, so the client can show a friendly "just taken" message without an
 * unhandled error boundary.
 */

import { revalidatePath } from 'next/cache'
import { requireRole } from '@/server/auth/session'
import { pickupsRepo } from '@/server/db/repositories/pickups'
import { createSignedUploadUrl } from '@/lib/storage'
import { geocodeAddress } from '@/lib/geocoding'
import {
  PickupService,
  createPickupSchema,
  updatePickupSchema,
  PickupAlreadyClaimedError,
} from '@/features/pickups'
import type { PickupStatus } from '@/config/constants'

// ─── Donor: Create pickup (DON-01, DON-03) ───────────────────────────────────

/**
 * Creates a new pickup request. Geocodes the address server-side (Nominatim, D-05).
 * Returns the new pickup's id on success.
 *
 * Caller (PickupForm) should handle the case where geo is null:
 * "Address not found — please drag the pin to set your location."
 * In that case pass lat/lng from the draggable Leaflet pin.
 */
export async function createPickupAction(formData: unknown): Promise<{
  success: true; pickupId: string
} | {
  success: false; error: string; fieldErrors?: Record<string, string[]>
}> {
  try {
    const session = await requireRole(['donor'])

    const parsed = createPickupSchema.safeParse(formData)
    if (!parsed.success) {
      return {
        success:     false,
        error:       'Validation failed',
        fieldErrors: parsed.error.flatten().fieldErrors as Record<string, string[]>,
      }
    }

    const { address, lat: formLat, lng: formLng, ...rest } = parsed.data

    // Geocode server-side (D-05) — never call Nominatim at render time
    let lat = formLat
    let lng = formLng

    if (lat === undefined || lng === undefined) {
      const geo = await geocodeAddress(address)
      if (!geo) {
        return {
          success: false,
          error: 'Address not found — please drag the pin to set your location.',
        }
      }
      lat = geo.lat
      lng = geo.lng
    }

    const pickup = await pickupsRepo.insert({
      ...rest,
      donorId:   session.userId,
      status:    'requested',
      address,
      lat:       lat!,
      lng:       lng!,
    })

    revalidatePath('/portal/board')
    revalidatePath('/portal/pickups')

    return { success: true, pickupId: pickup.id }
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Could not create pickup'
    return { success: false, error: message }
  }
}

// ─── Donor/Volunteer: Photo upload URL (DON-02, VOL-05, D-04) ────────────────

/**
 * Issues a signed upload URL for a pickup photo. Both donor (food photo) and
 * volunteer (proof-of-delivery) call this action.
 *
 * Object path: pickups/{pickupId}/{userId}-{timestamp}.{ext}
 * The path is constructed server-side — callers cannot choose an arbitrary path.
 *
 * The caller (PhotoUploader component) then:
 *   1. Compresses the image client-side (browser-image-compression, D-04)
 *   2. PUT to signedUrl directly (raw fetch — no supabase-js in browser bundle needed)
 *   3. Calls savePhotoPathAction with the objectPath to persist it in the DB
 */
export async function getSignedUploadUrlAction(
  pickupId: string,
  filename: string,
): Promise<{ signedUrl: string; token: string; objectPath: string }> {
  const session = await requireRole(['donor', 'volunteer'])

  // Verify the pickup exists (no IDOR — caller must have access to this pickup)
  const pickup = await pickupsRepo.getById(pickupId)
  if (!pickup) throw new Error('Pickup not found')

  // Donor can only upload food photo; volunteer can only upload proof photo.
  // Fine-grained ownership enforced in savePhotoPathAction after upload.
  const ext        = (filename.split('.').pop() ?? 'jpg').toLowerCase()
  const objectPath = `pickups/${pickupId}/${session.userId}-${Date.now()}.${ext}`

  return createSignedUploadUrl(objectPath)
}

/**
 * Save an uploaded photo's object path to the DB.
 * Called after a successful browser PUT to the signed upload URL.
 *
 * Determines which field to update based on the caller's role:
 *   - Donor → foodPhotoPath
 *   - Volunteer → proofPhotoPath
 */
export async function savePhotoPathAction(
  pickupId:   string,
  objectPath: string,
  photoType:  'food' | 'proof',
): Promise<void> {
  const session = await requireRole(['donor', 'volunteer'])

  if (photoType === 'food') {
    await PickupService.saveFoodPhotoPath(pickupId, session.userId, objectPath)
    revalidatePath(`/portal/pickups/${pickupId}`)
  } else {
    await PickupService.saveProofPhotoPath(pickupId, session.userId, objectPath)
    revalidatePath(`/portal/pickups/${pickupId}`)
    revalidatePath('/portal/board')
  }
}

// ─── Donor: Edit (DON-05) ─────────────────────────────────────────────────────

export async function editPickupAction(
  pickupId: string,
  formData: unknown,
): Promise<{ success: true } | { success: false; error: string; fieldErrors?: Record<string, string[]> }> {
  try {
    const session = await requireRole(['donor'])

    const parsed = updatePickupSchema.safeParse(formData)
    if (!parsed.success) {
      return {
        success:     false,
        error:       'Validation failed',
        fieldErrors: parsed.error.flatten().fieldErrors as Record<string, string[]>,
      }
    }

    // Re-geocode if address changed (D-05)
    const updates = { ...parsed.data }
    if (updates.address) {
      const geo = await geocodeAddress(updates.address)
      if (geo) {
        (updates as Record<string, unknown>).lat = geo.lat;
        (updates as Record<string, unknown>).lng = geo.lng
      }
      // If geo returns null, keep existing lat/lng from DB (editPickup will preserve them)
    }

    await PickupService.editPickup(pickupId, session.userId, parsed.data)

    revalidatePath(`/portal/pickups/${pickupId}`)
    revalidatePath('/portal/pickups')
    revalidatePath('/portal/board')

    return { success: true }
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Could not save changes'
    return { success: false, error: message }
  }
}

// ─── Donor: Cancel (DON-05, D-08, D-09) ──────────────────────────────────────

export async function cancelPickupAction(
  pickupId: string,
): Promise<{ success: true } | { success: false; error: string }> {
  try {
    const session = await requireRole(['donor'])
    await PickupService.cancelPickup(pickupId, session.userId)

    revalidatePath(`/portal/pickups/${pickupId}`)
    revalidatePath('/portal/pickups')
    revalidatePath('/portal/board')

    return { success: true }
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Could not cancel pickup'
    return { success: false, error: message }
  }
}

// ─── Donor: Quick-repost (DON-06, D-09) ──────────────────────────────────────

export async function repostPickupAction(
  originalPickupId: string,
): Promise<{ success: true; newPickupId: string } | { success: false; error: string }> {
  try {
    const session  = await requireRole(['donor'])
    const newId    = await PickupService.repostPickup(originalPickupId, session.userId)

    revalidatePath('/portal/pickups')
    revalidatePath('/portal/board')

    return { success: true, newPickupId: newId }
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Could not repost pickup'
    return { success: false, error: message }
  }
}

// ─── Volunteer: Claim (VOL-03, D-07) ─────────────────────────────────────────

/**
 * Atomic claim — returns a typed result instead of throwing.
 * The UI (ClaimButton) uses success:false + alreadyClaimed to show a friendly
 * "just taken" state rather than an unhandled error boundary.
 */
export async function claimPickupAction(
  pickupId: string,
): Promise<
  | { success: true;  pickupId: string }
  | { success: false; alreadyClaimed: true;  error: string }
  | { success: false; alreadyClaimed: false; error: string }
> {
  try {
    const session = await requireRole(['volunteer'])
    await PickupService.claimPickup(pickupId, session.userId)

    revalidatePath('/portal/board')
    revalidatePath(`/portal/pickups/${pickupId}`)
    revalidatePath('/portal/pickups')

    return { success: true, pickupId }
  } catch (err) {
    if (err instanceof PickupAlreadyClaimedError) {
      return {
        success:        false,
        alreadyClaimed: true,
        error:          err.message,
      }
    }
    const message = err instanceof Error ? err.message : 'Could not claim pickup'
    return { success: false, alreadyClaimed: false, error: message }
  }
}

// ─── Volunteer: Advance status (VOL-04, D-08) ─────────────────────────────────

export async function advanceStatusAction(
  pickupId: string,
  toStatus: string,   // validated as PickupStatus inside service layer
): Promise<{ success: true } | { success: false; error: string }> {
  try {
    const session = await requireRole(['volunteer'])
    await PickupService.advanceStatus(pickupId, toStatus as PickupStatus, session.userId)

    revalidatePath(`/portal/pickups/${pickupId}`)
    revalidatePath('/portal/board')
    revalidatePath('/portal/pickups')

    return { success: true }
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Could not advance status'
    return { success: false, error: message }
  }
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → server action | All input crosses this boundary — validated with Zod before any DB write |
| server action → PickupService | actorId always sourced from `requireRole()` return, never from request body |
| server action → Storage | objectPath constructed server-side; callers cannot choose arbitrary paths |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-13 | Elevation of Privilege | IDOR — caller supplies a `pickupId` they don't own | mitigate | Every write action → PickupService checks `donorId === actorId` or `volunteerId === actorId` before DB write; actorId comes from Clerk session only |
| T-02-14 | Tampering | Status machine bypass — client sends `toStatus='delivered'` directly | mitigate | `advanceStatusAction` passes `toStatus` to `PickupService.advanceStatus` which runs `validateTransition(pickup.status, toStatus)` — invalid transitions throw `InvalidTransitionError` |
| T-02-15 | Tampering | Signed URL path injection — client supplies crafted `objectPath` | mitigate | `getSignedUploadUrlAction` ignores any client-supplied path; `objectPath = pickups/${pickupId}/${session.userId}-${Date.now()}.ext` is constructed server-side entirely |
| T-02-16 | Tampering | `savePhotoPathAction` called with a path not owned by the caller | mitigate | Delegates to `PickupService.saveFoodPhotoPath` / `saveProofPhotoPath` which check `donorId`/`volunteerId === actorId` before updating DB |
| T-02-17 | Information Disclosure | `claimPickupAction` leaking other volunteers' state | accept | Returns a union type `{ success, alreadyClaimed, error }` — no pickup data from the competing volunteer is included in the response |
</threat_model>
