```typescript
  requested:  'requested',

    <span
      className={`inline-flex items-center gap-1.5 rounded-full px-2.5 py-0.5 text-xs font-bold ${className}`}
```

**File 1: `src/features/pickups/components/PhotoUploader.tsx`**

Client component: compress → PUT → save path. No supabase-js in browser (Option B from RESEARCH.md).

```typescript
'use client'

import { useState, useRef } from 'react'
import imageCompression from 'browser-image-compression'
import { getSignedUploadUrlAction, savePhotoPathAction } from '@/features/pickups'

interface Props {
  pickupId:  string
  photoType: 'food' | 'proof'
  onSuccess?: (objectPath: string) => void
}

const COMPRESSION_OPTIONS = {
  maxWidthOrHeight: 1600,
  maxSizeMB:        1,
  useWebWorker:     true,
  fileType:         'image/jpeg' as const,
}

export function PhotoUploader({ pickupId, photoType, onSuccess }: Props) {
  const [status, setStatus] = useState<'idle' | 'compressing' | 'uploading' | 'done' | 'error'>('idle')
  const [errorMsg, setErrorMsg] = useState<string | null>(null)
  const inputRef = useRef<HTMLInputElement>(null)

  async function handleFile(file: File) {
    setStatus('compressing')
    setErrorMsg(null)

    try {
      // Step 1: compress client-side (D-04)
      const compressed = await imageCompression(file, COMPRESSION_OPTIONS)

      // Step 2: get signed upload URL (server action)
      setStatus('uploading')
      const { signedUrl, objectPath } = await getSignedUploadUrlAction(pickupId, file.name)

      // Step 3: raw PUT to signed URL — no supabase-js needed in browser bundle
      const res = await fetch(signedUrl, {
        method:  'PUT',
        body:    compressed,
        headers: { 'Content-Type': 'image/jpeg' },
      })

      if (!res.ok) throw new Error(`Upload failed: ${res.status}`)

      // Step 4: save object path in DB (server action)
      await savePhotoPathAction(pickupId, objectPath, photoType)

      setStatus('done')
      onSuccess?.(objectPath)
    } catch (err) {
      setStatus('error')
      setErrorMsg(err instanceof Error ? err.message : 'Upload failed')
    }
  }

  return (
    <div className="space-y-2">
      <input
        ref={inputRef}
        type="file"
        accept="image/*"
        capture="environment"
        className="sr-only"
        onChange={(e) => {
          const file = e.target.files?.[0]
          if (file) handleFile(file)
        }}
      />
      {status === 'idle' && (
        <button
          type="button"
          onClick={() => inputRef.current?.click()}
          className="rj-press rounded-lg border border-dashed border-border px-4 py-3 text-sm text-muted-foreground w-full text-center hover:bg-surface-2 transition-colors"
        >
          {photoType === 'food' ? 'Add food photo (optional)' : 'Upload proof-of-delivery photo'}
        </button>
      )}
      {(status === 'compressing' || status === 'uploading') && (
        <div className="rj-skeleton h-12 rounded-lg" aria-label={`${status}…`} />
      )}
      {status === 'done' && (
        <p className="text-sm text-leaf font-medium">Photo uploaded</p>
      )}
      {status === 'error' && (
        <div className="space-y-1">
          <p className="text-sm text-danger">{errorMsg}</p>
          <button
            type="button"
            onClick={() => { setStatus('idle'); setErrorMsg(null) }}
            className="rj-press text-sm text-primary underline"
          >
            Try again
          </button>
        </div>
      )}
    </div>
  )
}
```

```typescript
'use client'

import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { useRouter } from 'next/navigation'
import { useState } from 'react'
import {
  createPickupSchema, updatePickupSchema,
  createPickupAction, editPickupAction,
} from '@/features/pickups'
import type { CreatePickupInput } from '@/features/pickups'
import { PickupMap } from './PickupMap'
import { PhotoUploader } from './PhotoUploader'
import type { Pickup } from '@/server/db/schema'
import { FOOD_CATEGORIES, QUANTITY_UNITS, PICKUP_ROUTES } from '@/config/constants'

const CATEGORY_LABELS: Record<string, string> = {
  cooked_meal: 'Cooked Meal', raw_produce: 'Raw Produce',
  packaged: 'Packaged Food', bakery: 'Bakery', other: 'Other',
}

interface Props {
  mode:    'create'
  pickupId?: undefined
  defaults?: undefined
}
interface EditProps {
  mode:    'edit'
  pickupId: string
  defaults: Partial<CreatePickupInput>
}

export function PickupForm({ mode, pickupId, defaults }: Props | EditProps) {
  const router  = useRouter()
  const schema  = mode === 'create' ? createPickupSchema : updatePickupSchema
  const [serverError, setServerError] = useState<string | null>(null)

  // Draggable pin state (D-05): starts at Ahmedabad center; updated by geocode result or pin drag
  const [pinLat, setPinLat] = useState<number | undefined>(defaults ? undefined : 23.0225)
  const [pinLng, setPinLng] = useState<number | undefined>(defaults ? undefined : 72.5714)
  const [newPickupId, setNewPickupId] = useState<string | undefined>(pickupId)

  const {
    register, handleSubmit, setValue, watch,
    formState: { errors, isSubmitting },
  } = useForm<CreatePickupInput>({
    resolver:      zodResolver(schema as typeof createPickupSchema),
    defaultValues: defaults ?? {},
  })

  async function onSubmit(data: CreatePickupInput) {
    setServerError(null)
    // Pass pin lat/lng if user has adjusted them (overrides geocode result)
    const payload = {
      ...data,
      lat: pinLat,
      lng: pinLng,
    }

    const result = mode === 'create'
      ? await createPickupAction(payload)
      : await editPickupAction(pickupId!, payload)

    if (!result.success) {
      setServerError(result.error)
      return
    }

    if (mode === 'create' && result.success) {
      setNewPickupId((result as { success: true; pickupId: string }).pickupId)
    }
    router.push(
      mode === 'create'
        ? PICKUP_ROUTES.donorDetail((result as { success: true; pickupId: string }).pickupId)
        : PICKUP_ROUTES.donorDetail(pickupId!),
    )
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-5 max-w-lg">

      {/* Food category (D-01) */}
      <div className="space-y-1">
        <label htmlFor="foodCategory" className="text-sm font-medium text-foreground">
          Food type <span className="text-danger">*</span>
        </label>
        <select
          id="foodCategory"
          {...register('foodCategory')}
          className={`block w-full h-[46px] rounded-[10px] border px-3 text-sm bg-surface
            ${errors.foodCategory ? 'border-danger rj-field--error' : 'border-border'}`}
        >
          <option value="">Select food type…</option>
          {FOOD_CATEGORIES.map((c) => (
            <option key={c} value={c}>{CATEGORY_LABELS[c]}</option>
          ))}
        </select>
        {errors.foodCategory && (
          <p className="text-xs text-danger">{errors.foodCategory.message as string}</p>
        )}
```

```typescript
import { useQuery } from '@tanstack/react-query'
import { QUERY_KEYS } from '@/config/constants'
import { pickupsRepo } from '@/server/db/repositories/pickups'

// Note: TanStack Query hooks call server actions in the queryFn.
// For RSC-first patterns, the list page fetches server-side; this hook
// is available for client-side refetch after mutations.
export function useMyPickups(userId: string) {
  return useQuery({
    queryKey: QUERY_KEYS.pickups.myPickups(userId),
    queryFn:  async () => {
      // This is a server-side call via TanStack Query in a client component.
      // In practice the donor list page is a RSC; this hook is used for optimistic
      // refresh after create/cancel/edit mutations.
      const res = await fetch(`/api/pickups/mine?userId=${userId}`)
      if (!res.ok) throw new Error('Failed to load pickups')
      return res.json() as Promise<Awaited<ReturnType<typeof pickupsRepo.getByDonorId>>>
    },
    staleTime: 0,
  })
}
```
