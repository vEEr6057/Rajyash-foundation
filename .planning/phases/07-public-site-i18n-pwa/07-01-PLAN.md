// src/i18n/request.ts
// SECURITY (T-7-01): locale validated against allowlist before use in file path resolution
import { cookies } from "next/headers";
import { getRequestConfig } from "next-intl/server";

export const SUPPORTED_LOCALES = ["en", "gu", "hi"] as const;
export type Locale = (typeof SUPPORTED_LOCALES)[number];

function isValidLocale(v: string | undefined): v is Locale {
  return SUPPORTED_LOCALES.includes(v as Locale);
}

export default getRequestConfig(async () => {
  const store = await cookies();
  const raw = store.get("NEXT_LOCALE")?.value;
  // Allowlist guard — invalid/missing cookie falls back to 'en' (no throw, mirrors middleware pattern)
  const locale: Locale = isValidLocale(raw) ? raw : "en";

  return {
    locale,
    // Merged namespace strategy: all 4 namespaces merged into one locale file for simplicity
    // (RESEARCH Open Question 2 — split only if files exceed ~200 KB, unlikely for this app)
    messages: {
      ...(await import(`./messages/${locale}/common.json`)).default,
      ...(await import(`./messages/${locale}/landing.json`)).default,
      ...(await import(`./messages/${locale}/portal.json`)).default,
      ...(await import(`./messages/${locale}/admin.json`)).default,
    },
  };
});
```

// src/features/public/actions/setLocale.ts
"use server";
// SECURITY (T-7-01): allowlist guard — only en/gu/hi accepted; any other value silently rejected
import { cookies } from "next/headers";
import { revalidatePath } from "next/cache";
import { SUPPORTED_LOCALES, type Locale } from "@/i18n/request";

export async function setLocaleCookieAction(locale: Locale): Promise<void> {
  if (!SUPPORTED_LOCALES.includes(locale)) return; // silently reject invalid values
  const store = await cookies();
  store.set("NEXT_LOCALE", locale, {
    path: "/",
    maxAge: 60 * 60 * 24 * 365, // 1 year — D-09: persists across sessions
    httpOnly: false, // RESEARCH Anti-Patterns: must NOT be httpOnly — readable by server action
    sameSite: "lax",
    secure: process.env.NODE_ENV === "production",
  });
  // Forces server components to re-run getRequestConfig with the new cookie (RESEARCH Pattern 2)
  revalidatePath("/", "layout");
}
```

Create `src/features/public/components/LanguageSwitcher.tsx`:
```tsx
"use client";
import { useRouter } from "next/navigation";
import { useTransition } from "react";
import { useLocale, useTranslations } from "next-intl";
import { cn } from "@/lib/utils";
import { setLocaleCookieAction } from "../actions/setLocale";
import type { Locale } from "@/i18n/request";

const LOCALE_LABELS: Record<Locale, string> = {
  en: "EN",
  gu: "ગુ",  // Gujarati script label
  hi: "हि",  // Devanagari script label
};

export function LanguageSwitcher({ className }: { className?: string }) {
  const router = useRouter();
  const locale = useLocale() as Locale; // RESEARCH Pitfall 3: use useLocale(), NOT navigator.language
  const t = useTranslations("common");
  const [isPending, startTransition] = useTransition();

  function handleLocaleChange(next: Locale) {
    if (next === locale) return; // already selected — no-op
    startTransition(async () => {
      await setLocaleCookieAction(next);
      router.refresh(); // re-runs Server Components with new NEXT_LOCALE cookie
    });
  }

  return (
    <div
      className={cn("flex items-center gap-1", className)}
      role="group"
      aria-label={t("nav.language")}
    >
      {(["en", "gu", "hi"] as const).map((l) => (
        <button
          key={l}
          type="button"
          onClick={() => handleLocaleChange(l)}
          disabled={isPending}
          aria-pressed={locale === l}
          className={cn(
            "rounded px-2 py-0.5 text-sm font-medium transition-colors",
            locale === l
              ? "bg-primary text-white"
              : "text-muted-foreground hover:text-foreground",
            "disabled:opacity-50 disabled:cursor-not-allowed",
          )}
        >
          {LOCALE_LABELS[l]}
        </button>
      ))}
    </div>
  );
}
```
