```javascript
// scripts/gen-icons.mjs — run once, then delete
import sharp from 'sharp';
import path from 'path';
import { fileURLToPath } from 'url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const publicDir = path.join(__dirname, '..', 'public');

// Saffron (#C04E12) background with cream (#FBF7F0) mark
// representing Rajyash Food Rescue brand — placeholder until user imports design
async function makeIcon(size, outputPath, withMaskablePadding = false) {
  const padding = withMaskablePadding ? Math.round(size * 0.1) : 0;
  const innerSize = size - padding * 2;

  const svg = `<svg xmlns="http://www.w3.org/2000/svg" width="${size}" height="${size}" viewBox="0 0 ${size} ${size}">
    <rect width="${size}" height="${size}" fill="#C04E12"/>
    <rect x="${padding}" y="${padding}" width="${innerSize}" height="${innerSize}" rx="${innerSize * 0.15}" fill="#C04E12"/>
    <text x="${size/2}" y="${size/2 + innerSize*0.12}" 
          font-family="sans-serif" font-weight="bold" 
          font-size="${innerSize * 0.45}" fill="#FBF7F0" 
          text-anchor="middle" dominant-baseline="middle">🍲</text>
    <text x="${size/2}" y="${size * 0.82}" 
          font-family="sans-serif" font-weight="600" 
          font-size="${innerSize * 0.12}" fill="#FBF7F0" 
          text-anchor="middle">FOOD RESCUE</text>
  </svg>`;

  await sharp(Buffer.from(svg)).resize(size, size).png().toFile(outputPath);
  console.log('Generated', outputPath);
}

await makeIcon(192, path.join(publicDir, 'icon-192.png'));
await makeIcon(512, path.join(publicDir, 'icon-512.png'));
await makeIcon(512, path.join(publicDir, 'icon-512-maskable.png'), true);
console.log('All icons generated. Replace with real brand assets when available.');
```
