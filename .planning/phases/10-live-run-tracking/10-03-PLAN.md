```typescript
// ── TRK-06: route + ETA to next pending stop ─────────────────────────────────
import { haversineMeters, estimateEtaMinutes, straightLineRoute } from "@/lib/routing";
import { fetchOsrmRoute } from "@/lib/routing.server";

/**
 * TRK-06: route polyline + haversine ETA from driver's current position to the
 * next pending stop on a run. Destination is resolved server-side from the DB
 * (never trust a client-supplied dest). Auth: admin or the run's assigned driver.
 * Falls back to a straight line + haversine when OSRM is unavailable.
 */
export async function getRunRoute(
  runId: string,
  fromLat: number,
  fromLng: number,
): Promise<Result<{ coords: [number, number][]; etaMinutes: number; source: "osrm" | "line" }>> {
  const session = await getSession();
  if (!session) return fail("UNAUTHORIZED", "Sign in first.");
  if (!Number.isFinite(fromLat) || !Number.isFinite(fromLng)) {
    return fail("VALIDATION", "Invalid coordinates.");
  }

  const run = await runsRepo.getById(runId);
  if (!run) return fail("NOT_FOUND", "Run not found.");

  const allowed = session.role === "admin" || run.driverId === session.userId;
  if (!allowed) return fail("FORBIDDEN", "Not authorised.");

  const allStops = await runStopsRepo.getByRunId(runId);
  // Next stop: lowest seq among pending stops that have a geocoded lat/lng
  const nextStop = allStops
    .filter((s) => s.status === "pending" && s.lat !== null && s.lng !== null)
    .sort((a, b) => a.seq - b.seq)[0];

  if (!nextStop || nextStop.lat === null || nextStop.lng === null) {
    return fail("NOT_FOUND", "No pending stop with a location found.");
  }

  const from = { lat: fromLat, lng: fromLng };
  const to = { lat: nextStop.lat, lng: nextStop.lng };

  const osrm = await fetchOsrmRoute(from, to);
  if (osrm) {
    const etaMinutes = estimateEtaMinutes(
      haversineMeters(from, to), // use haversine for ETA even with OSRM route (consistent)
    );
    return { ok: true, coords: osrm.coords, etaMinutes, source: "osrm" };
  }

  const coords = straightLineRoute(from, to);
  const etaMinutes = estimateEtaMinutes(haversineMeters(from, to));
  return { ok: true, coords, etaMinutes, source: "line" };
}
```

```typescript
"use client";

import { useTranslations } from "next-intl";
import { MapView } from "@/features/pickups/components/MapView";
import { useLiveRunLocation } from "@/features/runs/hooks/useLiveRunLocation";
import { useRunRoute } from "@/features/runs/hooks/useRunRoute";

/**
 * Coordinator/driver run watcher (TRK-05/06). Mirrors LiveTrackingMap exactly:
 * live driver marker, route polyline (--route gold), ETA card, stale indicator.
 * nextStop = next pending stop's lat/lng (resolved by the page from runWithStops).
 */
export function RunLiveMap({
  runId,
  active,
  nextStop,
}: {
  runId: string;
  active: boolean;
  nextStop: { lat: number; lng: number } | null;
}) {
  const t = useTranslations("portal");
  const { position, stale, lastSeen, connection } = useLiveRunLocation({ runId, active });
  const live = position ? { lat: position.lat, lng: position.lng } : null;
  const { route, etaMinutes, source } = useRunRoute({ runId, live, active });

  return (
    <div className="space-y-2">
      <MapView
        markers={[]}
        destination={nextStop}
        live={live}
        liveStale={stale}
        route={route}
        height={280}
      />
      {etaMinutes != null && !stale && (
        <div className="flex items-center justify-between rounded-lg border border-border bg-surface px-3 py-2 text-sm font-semibold text-leaf">
          <span>{t("run.liveMap.etaAway", { minutes: etaMinutes })}</span>
          {source === "line" && (
            <span className="text-xs font-normal text-muted-foreground">
              {t("run.liveMap.etaEstimate")}
            </span>
          )}
        </div>
      )}
      <div className="flex items-center justify-between text-xs">
        <span className="flex items-center gap-1.5">
          {!stale && position ? (
            <span className="rj-dot-live inline-block size-2 rounded-full bg-leaf-bright" aria-hidden />
          ) : (
            <span className="inline-block size-2 rounded-full bg-muted-foreground/50" aria-hidden />
          )}
          <span className={stale ? "text-muted-foreground" : "text-leaf"}>
            {position
              ? stale
                ? t("run.liveMap.stale")
                : t("run.liveMap.live")
              : t("run.liveMap.waiting")}
          </span>
        </span>
        {lastSeen && <span className="text-muted-foreground">updated {lastSeen}</span>}
        {connection === "polling" && (
          <span className="text-muted-foreground">{t("run.liveMap.reconnecting")}</span>
        )}
      </div>
    </div>
  );
}
```
